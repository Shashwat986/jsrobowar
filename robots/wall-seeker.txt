#Energy 150
#Damage 150
#Shields 0
#Processor 15
#Normal bullets
Start:
    -5 speedx' store
Loop:
    x 30 < AtWall ifg
    Loop jump

AtWall:
    0 speedx' store
SetAim:
    0 aim' store
WallLoop:
    aim 5 + aim' store
    range 0 > Shoot if
    aim 180 > SetAim WallLoop ifeg

Shoot:
    energy fire' sto
    return
